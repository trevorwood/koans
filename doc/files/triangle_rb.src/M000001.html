def triangle(a, b, c)

if ((a<=0)||(b<=0)||(c<=0)) 
 raise TriangleError, "A triangle cannot have a side of length 0"
elsif ((a+b<=c)||(b+c<=a)||(a+c<=b))
 raise TriangleError, "The length of two sides can't be smaller than the third side"
end



if ((a==0)||(b==0)||(c==0)) 
  raise TriangleError, "Triangle cannot have a side of length 0"
 end
if ((a==b) && (b==c))
  :equilateral
 elsif (((a==b)&&(a!=c)) || ((b==c)&&(b!=a)) || ((c==a)&&(c!=b)))
  :isosceles
 else

 end

end

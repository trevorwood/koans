def score(dice)
  # You need to write this method
 points = 0 
 sum = 0

 ones = dice.select {|die| die==1}
 twos = dice.select {|die| die == 2}
 threes = dice.select {|die| die ==3}
 fours = dice.select {|die| die == 4}
 fives = dice.select {|die| die == 5}
 sixes = dice.select {|die| die == 6}
 
 dice.each {|die|
  if(die==5) 
   sum += 50 
  elsif(die==1)
    sum += 100
  else
   sum += 0
  end
 }

 if(ones.size >=3) 
  sum += 700
 end
 if(twos.size >= 3)
  sum += 200
 end
 if(threes.size >= 3)
  sum += 300
 end
 if(fours.size >=3)
  sum += 400
 end
 if(fives.size >=3) 
  sum += 350
 end
 if(sixes.size >= 3)
  sum += 600
 end


 points = sum 
end
